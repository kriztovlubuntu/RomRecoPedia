Herramienta de Debloat Automatizada (Python)

Este script escanea un directorio de una ROM extraída (por ejemplo, /system/app o /system/priv-app) y remueve las carpetas de bloatware predefinidas para aligerar el sistema de un plumazo.

import os
import shutil

# Lista de aplicaciones basura comunes a remover
BLOATWARE_LIST = [
    "Facebook", "FacebookAppManager", "FacebookServices",
    "YouTube", "Gmail2", "Maps", "Duo", "Photos",
    "Chrome", "Velvet", "CloudPrint", "Music2", "Videos"
]

def clean_rom_folder(target_dir):
    print(f"[*] Escaneando directorio: {target_dir}")
    if not os.path.exists(target_dir):
        print("[-] Error: El directorio no existe.")
        return

    removed_count = 0
    for folder_name in os.listdir(target_dir):
        if folder_name in BLOATWARE_LIST:
            full_path = os.path.join(target_dir, folder_name)
            try:
                shutil.rmtree(full_path)
                print(f"[✓] Eliminado con éxito: {folder_name}")
                removed_count += 1
            except Exception as e:
                print(f"[-] Error al eliminar {folder_name}: {e}")
                
    print(f"\n[+] Proceso terminado. Se removieron {removed_count} carpetas de basura.")

if __name__ == "__main__":
    print("=== OASIS ROM DEBLOATER ===")
    ruta_rom = input("Introduce la ruta de tu carpeta /app o /priv-app extraída: ").strip()
    clean_rom_folder(ruta_rom)

Cómo usar: Abre Thonny en Zorin OS, pega este código, guárdalo como debloat.py y ejecútalo indicando la carpeta a limpiar.
